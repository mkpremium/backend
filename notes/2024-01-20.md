```
SELECT count(*), min(worksheet.viewedAt), max(worksheet.viewedAt)
FROM mkpremium worksheet

WHERE worksheet._documentType = 'worksheet'
  AND worksheet.status IN ['OPEN', 'LOOKING_MEETING']
AND worksheet.queueId IS NULL
AND worksheet.buildingAddress.province IN ["MADRID","BARCELONA","ILLES BALEARS","VALENCIA"]
```

```
SELECT
  count(*) as count,
  min(worksheet.statusChangedAt) minStatusChangedAt,
  max(worksheet.statusChangedAt) maxStatusChangedAt,
  min(worksheet.viewedAt) minViewAt,
  max(worksheet.viewedAt) maxViewAt
FROM mkpremium worksheet

WHERE worksheet._documentType = 'worksheet'
  AND worksheet.status IN ['NO_SALE', 'LOOKING_MEETING']
  and statusChangedAt IS NOT NULL
  AND inFreezer = true
  
AND worksheet.buildingAddress.province IN ["MADRID","BARCELONA","ILLES BALEARS","VALENCIA"]
```
